ACTOR GrenadeLauncher : Postal3dWeaponBase
{

   States
   {
	Spawn:
		GLAP A -1
		Stop
	Ready: 
		TNT1 A 0 A_JumpIfInventory("Kicking",1,"Kick")
		CLGL A 1 A_WeaponReady
		Loop
	Deselect:
		CLGL A 1 A_Lower
		TNT1 AA 0 A_Lower
		Loop
	Select:
		CLGL A 1 A_Raise
		TNT1 AA 0 A_Raise
		LOOP
	Fire: 
		CLGL A 1
		Goto Ready
  }
}
